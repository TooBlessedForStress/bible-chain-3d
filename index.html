async function createNewBlock() { if (!walletPublicKey) { alert("Please connect your wallet first"); return; } const version = document.getElementById('version-select').value; const verse = await getSecureRandomVerse(version); // Build a clear, safe transaction const tx = new solanaWeb3.Transaction(); const memo = `VERSECHAIN: Minting ${verse.ref} as validator • +10 HOLY`; tx.add( new solanaWeb3.TransactionInstruction({ keys: [], programId: new solanaWeb3.PublicKey("MemoSq4gqABAXKb96qnH8TysNcWxMyWCqXgDLGmfcHr"), data: new TextEncoder().encode(memo) // ← Fixed for browser }) ); const { blockhash } = await connection.getLatestBlockhash(); tx.recentBlockhash = blockhash; tx.feePayer = walletPublicKey; // Simulate first (security best practice) const simulation = await connection.simulateTransaction(tx); if (simulation.value.err) { alert("Transaction simulation failed. Please try again later."); return; } try { const signedTx = await window.solana.signTransaction(tx); const signature = await connection.sendRawTransaction(signedTx.serialize()); await connection.confirmTransaction(signature, "confirmed"); alert(`✅ Block successfully minted on Solana Devnet!\n\nVerse: ${verse.ref}\n+10 HOLY rewarded to your wallet\n\nView on Solscan: https://solscan.io/tx/${signature}?cluster=devnet`); // Add to 3D chain const newBlock = { id: Date.now(), reference: verse.ref, text: verse.text, validator: walletPublicKey.toBase58() }; if (blocks.length >= MAX_BLOCKS) { const oldest = blocks.shift(); if (oldest.mesh) scene.remove(oldest.mesh); } blocks.push(newBlock); createBlockMesh(newBlock, blocks.length - 1); repositionAllBlocks(); currentIndex = blocks.length - 1; focusBlock(currentIndex); // Update stats document.getElementById('verses-created').innerText = parseInt(document.getElementById('verses-created').innerText) + 1; document.getElementById('total-tokens').innerText = (parseInt(document.getElementById('total-tokens').innerText) + 10) + " HOLY"; } catch (err) { console.error(err); alert("Transaction was cancelled or failed."); } }
